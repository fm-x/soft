JUDE API User Guide

[Easy Merge]

Easy Merge function of JUDE API is not available in JUDE/Community.

Easy Merge

Reference projects can be added as follows:

  1. Open a project file.
  2. Easy Merge.
  3. Save as a JUDE project file.

import com.change_vision.jude.api.inf.project.ProjectAccessor;
import com.change_vision.jude.api.inf.project.ProjectAccessorFactory;

public class Test {
  public static void main(String[] args) {
    try {
        ProjectAccessor prjAccessor = ProjectAccessorFactory.getProjectAccessor();
        
        prjAccessor.open("C:\\API\\pro305.jude");
        prjAccessor.easyMerge("C:\\API\\pro305_ref.jude", false);
        
        prjAccessor.saveAs("C:\\API\\pro305_2.jude"); 
        
        prjAccessor.close();
        
    } catch (Exception e) {
        e.printStackTrace();
    }
  }
}
